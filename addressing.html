Уникальные названия линий

HTML

<div class="grid unique-line-names">
    <div>1</div>
    <div>2</div>
    <div>3</div>
    <div>4</div>
    <div>5</div>
    <div>6</div>
</div>

CSS

    .grid {
        display: grid;
    }
    .unique-line-names {
        grid-template-columns:
        10px [col1-start] 1fr [col1-end]
        10px [col2-start] 1fr [col2-end]
        10px [col3-start] 1fr [col3-end]
        10px [col4-start] 1fr [col4-end]
        10px;
    }
    .unique-line-names > div:nth-child(3n + 1) {
        grid-column: col1-start;
    }
    .unique-line-names > div:nth-child(3n + 2) {
        grid-column: col3-start;
    }
    .unique-line-names > div:nth-child(3n) {
        grid-column: col4-start;
    }
